git init - инициализация репозитория
git add file - добавление файла в область полготовки файлов
git status - проверка статуса репозитория
git commit -m "AAA" - создание коммита: внесение изменений  оставление комментария
git rm - убрать из версионного контроля
git config user.name "AAA"
git config user.email "AAA"  -  подпись(имя и почта), нужная перед добавдением комментария
git log - история коммитов
git branch AAA - создание ветки AAA
git checkout AAA - переключение на ветку AAA
git merge AAA - слияние текущей ветки с AAA
git branch -d AAA - закрытие ветки AAA
git diff - показать изменения после последнего коммита
git clone <ссылка> - копирование репозитория по ссылке
git show <хэш коммита> - просмотр полного списка изменений коммита
